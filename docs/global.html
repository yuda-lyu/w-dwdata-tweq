Example
import WDwdataTweq from './src/WDwdataTweq.mjs'

let yearStart = 2022
let yearEnd = 2022
let opt = {}
let ev = await WDwdataTweq(yearStart, yearEnd, opt)
    .catch((err) => {
        console.log(err)
    })
ev.on('change', (msg) => {
    delete msg.type
    console.log('change', msg)
})
